\begin{frame}{Current Claim}
  \begin{block}{What DRIP is now}
    DRIP is a cache-management method for under-covered multi-hop supports:
    \[
      \text{resident/easy } A \rightarrow \text{missing hidden support } B.
    \]
  \end{block}

  \begin{itemize}
    \item ESC discovers missing support $B$ conditioned on a resident or easy anchor $A$.
    \item Pair Lease keeps $A$ and $B$ resident as a support unit.
    \item PPR, role reranking, DRF/hubness, and redundancy penalties are not the current main method.
  \end{itemize}
\end{frame}

\begin{frame}{Two Branches, Different Roles}
  \begin{table}

  \begin{tabular}{lll}
    \toprule
    Branch & Purpose & Current reading \\
    \midrule
    Single-hop query shift & Calibration & LRU/FIFO are strong when evidence is directly visible \\
    No-shift multi-hop & Main method test & ESC + Pair Lease improves hidden-support residency \\
    \bottomrule
  \end{tabular}
  \end{table}

  \begin{alertblock}{Do not mix them}
    Query shift and multi-hop hidden evidence are orthogonal. The current
    validated contribution is no-topic-shift multi-hop cache residency.
  \end{alertblock}
\end{frame}

\begin{frame}{DRIP Formula}
  \begin{block}{Evidence-conditioned support completion}
    \[
      \esc(q,B \mid A)
      = \mathrm{sim}(\phi(q,A),B)\cdot \mathrm{link}(A,B)\cdot \mathrm{cue}(q,B).
    \]
  \end{block}

  \begin{block}{Demand and pair lease}
    \[
      D_{\mathrm{dir},t}(d) \leftarrow \text{direct dense evidence},
      \qquad
      D_{\mathrm{brg},t}(d) \leftarrow \esc(q_t,d).
    \]
    \[
      L_t(A),L_t(B) \leftarrow \rho L_{t-1} + \esc(q_t,B\mid A).
    \]
  \end{block}

  \begin{block}{Resident priority}
    \[
      P_t(d)=S_t(d)+D_{\mathrm{dir},t}(d)+D_{\mathrm{brg},t}(d)+\lambda_{\mathrm{pair}}L_t(d).
    \]
  \end{block}
\end{frame}

\begin{frame}{Admission Rule}
  \begin{block}{Direct-first writer}
    \[
      \text{admit } c \text{ over } v
      \Longleftrightarrow
      D(c) > m_r P_t(v).
    \]
  \end{block}

  \begin{enumerate}
    \item Direct candidates use the window write budget first.
    \item Bridge candidates use leftover budget.
    \item Bridge eviction avoids direct-protected, recently served, or direct-demand resident docs when possible.
  \end{enumerate}

  \begin{alertblock}{Reason}
    Hidden $B$ should repair an under-covered support set, not evict the visible
    anchor $A$ or other query-serving evidence.
  \end{alertblock}
\end{frame}

\begin{frame}{Final Story}
  \begin{enumerate}
    \item Single-hop query shift calibrates the boundary: recency baselines are strong.
    \item The real gap is hidden multi-hop support residency.
    \item DRIP repairs that gap with ESC and Pair Lease.
    \item Current positive evidence: 2Wiki and MuSiQue no-topic-shift multi-hop.
    \item Topic shift + multi-hop is the next combined benchmark, not the current main claim.
  \end{enumerate}

  \begin{block}{One sentence}
    DRIP is evidence-conditioned support completion for multi-hop cache residency.
  \end{block}
\end{frame}
